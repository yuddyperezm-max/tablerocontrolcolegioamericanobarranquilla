function calcPct(items) {
  if (!items.length) return 0;
  return Math.round(items.filter(i => i.done).length / items.length * 100);
}

function makeRingSVG(pct, color, size=52) {
  const r = 20, cx = size/2, cy = size/2;
  const circ = 2 * Math.PI * r;
  const dash = (pct / 100) * circ;
  return `<svg width="${size}" height="${size}" viewBox="0 0 ${size} ${size}">
    <circle cx="${cx}" cy="${cy}" r="${r}" fill="none" stroke="#e5e9f0" stroke-width="5"/>
    <circle cx="${cx}" cy="${cy}" r="${r}" fill="none" stroke="${color}" stroke-width="5"
      stroke-dasharray="${dash.toFixed(1)} ${(circ-dash).toFixed(1)}"
      stroke-dashoffset="${(circ/4).toFixed(1)}"
      stroke-linecap="round" transform="rotate(-90 ${cx} ${cy})"/>
  </svg>`;
}

function statusColor(st) {
  if (st==='en-curso') return '#EC8906';
  if (st==='proxima') return '#197AA4';
  return '#aaa';
}
function statusLabel(st) {
  if (st==='en-curso') return 'En curso';
  if (st==='proxima') return 'Próxima';
  return 'Pendiente';
}

let activeComp = null, activeFase = null;

function globalPct() {
  const all = [...componentes.map(c=>c.items), ...fases.map(f=>f.items)].flat();
  return Math.round(all.filter(i=>i.done).length / all.length * 100);
}

function renderMetrics() {
  const totalItems = componentes.reduce((s,c)=>s+c.items.length,0);
  const doneItems = componentes.reduce((s,c)=>s+c.items.filter(i=>i.done).length,0);
  const fasesDone = fases.filter(f=>calcPct(f.items)===100).length;
  const metrics = [
    { val: Math.round(doneItems/totalItems*100)+'%', label: 'Avance componentes' },
    { val: doneItems+'/'+totalItems, label: 'Acciones completadas' },
    { val: fasesDone+'/6', label: 'Fases completadas' },
    { val: 'Sep 2027', label: 'Meta radicación' },
  ];
  const grid = document.getElementById('metricsGrid');
  grid.innerHTML = '';
  metrics.forEach(m => {
    const box = document.createElement('div');
    box.className = 'metric-box';
    box.innerHTML = `<div class="m-val">${m.val}</div><div class="m-label">${m.label}</div>`;
    grid.appendChild(box);
  });
}

function renderGlobal() {
  const pct = globalPct();
  document.getElementById('globalPctTxt').textContent = pct+'%';
  document.getElementById('globalBar').style.width = pct+'%';
}

function renderRings() {
  const grid = document.getElementById('ringsGrid');
  grid.innerHTML = '';
  componentes.forEach((c, ci) => {
    const pct = calcPct(c.items);
    const card = document.createElement('div');
    card.className = 'ring-card' + (activeComp===ci?' active':'');
    card.innerHTML = makeRingSVG(pct, c.estadoColor) +
      `<div class="ring-pct">${pct}%</div>` +
      `<div class="ring-label">${c.label}</div>` +
      `<div class="ring-estado" style="background:${c.estadoColor}22;color:${c.estadoColor}">${c.estado}</div>`;
    card.onclick = () => { activeComp = activeComp===ci ? null : ci; renderRings(); renderCompDetail(); };
    grid.appendChild(card);
  });
}

function renderCompDetail() {
  const panel = document.getElementById('compDetail');
  const title = document.getElementById('compDetailTitle');
  const list = document.getElementById('compDetailList');
  if (activeComp===null) { panel.classList.remove('open'); return; }
  const c = componentes[activeComp];
  panel.classList.add('open');
  const done = c.items.filter(i=>i.done).length;
  title.textContent = c.label + ' · ' + done + '/' + c.items.length + ' acciones completadas · Estado: ' + c.estado;
  list.innerHTML = '';
  c.items.forEach(item => {
    const li = document.createElement('li');
    const chk = document.createElement('div');
    chk.className = 'chk' + (item.done?' done':'');
    // Solo visual — sin onclick
    li.appendChild(chk);
    const txt = document.createElement('span');
    txt.textContent = item.text;
    if (item.done) txt.style.color = '#888';
    li.appendChild(txt);
    list.appendChild(li);
  });
}
